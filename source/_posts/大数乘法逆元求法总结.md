3.  拓展欧几里得算法
```cpp
//扩展欧几里德  
void exp_gcd( LL a ,LL b,LL &x,LL &y) {  
     if( b == 0 ) {  
         x = 1;  
         y = 0;  
     }  
     else {  
          exp_gcd( b,a%b,x,y );  
          LL t;  
          t = x;  
          x = y;  
          y = t - a/b*y;  
     }  
}  
//逆元  
inline LL getNN(LL x) {  
        LL now , t;  
        exp_gcd( x, M,now,t );  
        return (now%M+M)%M;  
}  
```

4.  拓展欧几里得（简洁版）
```cpp
int64 inv(int64 x) {    
    //简洁版求逆元    
    if(x == 1) return 1;    
    return  inv(MOD%x) * (MOD - MOD/x) % MOD;    
} 
```
